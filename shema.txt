project-root/
├── bot/                     
│   ├── __init__.py        # Инициализация пакета бота
│   ├── core.py            # Основная логика бота (подключение к API, обработка команд)
│   ├── handlers.py        # Обработчики команд (например, для показа графика или замены смен)
│   └── utils.py           # Вспомогательные функции (например, для форматирования сообщений)
│
├── excel_parser/          
│   ├── __init__.py        # Инициализация пакета для работы с Excel
│   └── parser.py          # Логика парсинга Excel-таблицы с графиком дежурств
│
├── schedule/              
│   ├── __init__.py        # Инициализация пакета для работы с расписанием
│   └── manager.py         # Логика управления сменами и заменами (обновление, валидация данных)
│
├── config.py              # Конфигурация приложения (настройки, константы, пути к файлам)
├── .env                   # Файл с переменными окружения (например, API-токен бота)
├── requirements.txt       # Список зависимостей проекта (python-dotenv, Pillow, openpyxl/pandas и т.д.)
├── .gitignore             # Файл исключений для git (например, .env, __pycache__, venv и т.д.)
├── README.md              # Документация проекта (описание, инструкция по запуску, использование)
└── main.py                # Точка входа в приложение; запускает бота, загружает конфигурацию и т.п.
